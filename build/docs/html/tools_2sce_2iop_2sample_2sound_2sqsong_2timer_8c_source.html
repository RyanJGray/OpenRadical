24/* 4167 micro sec. = 1000 * 1000 / 240 = 1/240 sec */
25#define ONE_240TH 4167

27typedef struct TimerCtx {

33/* ------------------------------------------------ */
34/* The handler for timer interrupt */

36/* ------------------------------------------------ */
37unsigned int
38timer_handler (void *common)

40 TimerCtx *tc = (TimerCtx *) common;

42 iWakeupThread (tc->thread_id); // wakeup atick()

47/* ------------------------------------------------ */
48/* Configuring timer environment */
49/* タイマー設定 */
50/* ------------------------------------------------ */
51int
52set_timer (TimerCtx *timer)
53{
54 struct SysClock clock;
55 int timer_id;
56
57 /* 1/240 sec = ??? sysclock */
58 USec2SysClock (ONE_240TH, & clock);
59 timer->count = clock.low; /* within 32-bit */
60
61 /* Use sysclock timer */
62 if ((timer_id = AllocHardTimer (TC_SYSCLOCK, 32, 1)) <= 0) {
63 printf ("Can NOT allocate hard timer ...\n");
64 return (-1);
65 }
66 timer->timer_id = timer_id;
67
68 if (SetTimerHandler (timer_id, timer->count,
69 timer_handler, (void *) timer) != KE_OK) {
70 printf ("Can NOT set timeup timer handler ...\n");
71 return (-1);
72 }
73
74 if (SetupHardTimer (timer_id, TC_SYSCLOCK, TM_NO_GATE, 1) != KE_OK) {
75 printf ("Can NOT setup hard timer ...\n");
76 return (-1);
77 }
78
79 if (StartHardTimer (timer_id) != KE_OK) {
80 printf ("Can NOT start hard timer ...\n");
81 return (-1);
82 }
83
84 return 0;
85}
