17/*
18 * typedef char S8; 8bit signed
19 * typedef unsigned char U8; 8bit unsigned
20 * typedef short S16; 16bit signed
21 * typedef unsigned short U16; 16bit unsigned
22 * typedef int S32; 32bit signed
23 * typedef unsigned int U32; 32bit unsigned
24 *
25 * ファイルオフセット(ファイルの先頭からのバイト数)
26 * typedef U32 FOFST;
27 * #define NO_FOFST ((FOFST)-1)
28 * チャンクオフセット(チャンクの先頭からのバイト数)
29 * typedef U32 COFST;
30 * #define NO_COFST ((COFST)-1)
31 * チャンク内のオフセット(該当チャンクの先頭からのbyte数)
32 * typedef U32 SOFST;
33 * #define NO_SOFST ((SOFST)-1)
34 * ポジションオフセット(チャンク内の現在の位置からのバイト数)
35 * typedef U32 POFST;
36 * #define NO_POFST ((POFST)-1)
37 * チャンク内の現在の位置からのオフセット
38 * typedef U32 TOFST;
39 * #define NO_TOFST ((TOFST)-1)
40 * チャンクインデックス(チャンク内の構造体へのインデックス番号)
41 * typedef U16 CINDEX;
42 * #define NO_CINDEX ((CINDEX)-1)
43 * チャンク内の構造体へのインデックスナンバー
44 * typedef U16 SINDEX;
45 * #define NO_SINDEX ((SINDEX)-1)
46 */
