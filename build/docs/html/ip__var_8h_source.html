41struct sceEENetIpstat {
42 u_quad_t ips_total; /* total packets received */
43 u_quad_t ips_badsum; /* checksum bad */
44 u_quad_t ips_tooshort; /* packet too short */
45 u_quad_t ips_toosmall; /* not enough data */
46 u_quad_t ips_badhlen; /* ip header length < data size */
47 u_quad_t ips_badlen; /* ip length < ip header length */
48 u_quad_t ips_fragments; /* fragments received */
49 u_quad_t ips_fragdropped; /* frags dropped (dups, out of space) */
50 u_quad_t ips_fragtimeout; /* fragments timed out */
51 u_quad_t ips_forward; /* packets forwarded */
52 u_quad_t ips_fastforward; /* packets fast forwarded */
53 u_quad_t ips_cantforward; /* packets rcvd for unreachable dest */
54 u_quad_t ips_redirectsent; /* packets forwarded on same net */
55 u_quad_t ips_noproto; /* unknown or unsupported protocol */
56 u_quad_t ips_delivered; /* datagrams delivered to upper level*/
57 u_quad_t ips_localout; /* total ip packets generated here */
58 u_quad_t ips_odropped; /* lost packets due to nobufs, etc. */
59 u_quad_t ips_reassembled; /* total packets reassembled ok */
60 u_quad_t ips_fragmented; /* datagrams sucessfully fragmented */
61 u_quad_t ips_ofragments; /* output fragments created */
62 u_quad_t ips_cantfrag; /* don't fragment flag was set, etc. */
63 u_quad_t ips_badoptions; /* error in option processing */
64 u_quad_t ips_noroute; /* packets discarded due to no route */
65 u_quad_t ips_badvers; /* ip version != 4 */
66 u_quad_t ips_rawout; /* total raw ip packets generated */
67 u_quad_t ips_badfrags; /* malformed fragments (bad length) */
68 u_quad_t ips_rcvmemdrop; /* frags dropped for lack of memory */
69 u_quad_t ips_toolong; /* ip length > max ip packet size */
70 u_quad_t ips_nogif; /* no match gif found */
71};
