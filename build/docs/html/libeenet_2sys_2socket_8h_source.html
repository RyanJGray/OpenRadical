350/*
351 * Alignment requirement for CMSG struct manipulation.
352 * This basically behaves the same as ALIGN() ARCH/include/param.h.
353 * We declare it separately for two reasons:
354 * (1) avoid dependency between machine/param.h, and (2) to sync with kernel's
355 * idea of ALIGNBYTES at runtime.
356 * without (2), we can't guarantee binary compatibility in case of future
357 * changes in ALIGNBYTES.
358 */
359#define SCE_EENET___CMSG_ALIGN(n) (((n) + __sceEENetCmsgAlignbytes()) & ~__sceEENetCmsgAlignbytes())
360
361/* given pointer to struct cmsghdr, return pointer to next cmsghdr */
362#define SCE_EENET_CMSG_NXTHDR(mhdr, cmsg) \
363 (((caddr_t)(cmsg) + SCE_EENET___CMSG_ALIGN((cmsg)->cmsg_len) + \
364 SCE_EENET___CMSG_ALIGN(sizeof(struct sceEENetCmsghdr)) > \
365 (((caddr_t)(mhdr)->msg_control) + (mhdr)->msg_controllen)) ? \
366 (struct sceEENetCmsghdr *)NULL : \
367 (struct sceEENetCmsghdr *)((caddr_t)(cmsg) + SCE_EENET___CMSG_ALIGN((cmsg)->cmsg_len)))
368
369#define SCE_EENET_CMSG_FIRSTHDR(mhdr) ((struct sceEENetCmsghdr *)(mhdr)->msg_control)

371#define SCE_EENET_CMSG_SPACE(l) (SCE_EENET___CMSG_ALIGN(sizeof(struct sceEENetCmsghdr)) + __CMSG_ALIGN(l))
