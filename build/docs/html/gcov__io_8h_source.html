34/* Store a portable representation of VALUE in DEST using BYTES*8-1 bits.
35 Return a non-zero value if VALUE requires more than BYTES*8-1 bits
36 to store. */
37
38static int
39__store_long (long value, char *dest, size_t bytes)
40{
41 int upper_bit = (value < 0 ? 128 : 0);
42 size_t i;
43
44 if (value < 0) {
45 long oldvalue = value;
46 value = -value;
47 if (oldvalue != -value)
48 return 1;
49 }
50
51 for(i = 0 ; i < (sizeof (value) < bytes ? sizeof (value) : bytes) ; i++) {
52 dest[i] = value & (i == (bytes - 1) ? 127 : 255);
53 value = value / 256;
54 }
55
56 if (value && value != -1)
57 return 1;
58
59 for(; i < bytes ; i++)
60 dest[i] = 0;
61 dest[bytes - 1] |= upper_bit;
62 return 0;
63}

65/* Retrieve a quantity containing BYTES*8-1 bits from SOURCE and store
66 the result in DEST. Returns a non-zero value if the value in SOURCE
67 will not fit in DEST. */
68
69static int
70__fetch_long ( long *dest, char *source, size_t bytes )
71{
72 long value = 0;
73 int i;
74
75 for (i = bytes - 1; (size_t) i > (sizeof (*dest) - 1); i--)
76 if (source[i] & ((size_t) i == (bytes - 1) ? 127 : 255 ))
77 return 1;
78
79 for (; i >= 0; i--)
80 value = value * 256 + (source[i] & ((size_t)i == (bytes - 1) ?
81 127 : 255));
82
83 if ((source[bytes - 1] & 128) && (value > 0))
84 value = - value;
85
86 *dest = value;
87 return 0;
88}

90/* Write a BYTES*8-bit quantity to FILE, portably. Returns a non-zero
91 value if the write fails, or if VALUE can't be stored in BYTES*8
92 bits.
93
94 Note that VALUE may not actually be large enough to hold BYTES*8
95 bits, but BYTES characters will be written anyway.
96
97 BYTES may be a maximum of 10. */
98
99static int
100__write_long (long value, int file, size_t bytes )
101{
102 char c[10];
103
104 if (bytes > 10 || __store_long (value, c, bytes))
105 return 1;
106 else
107 return write(file, c, bytes) != bytes;
108}

110/* Read a quantity containing BYTES bytes from FILE, portably. Return
111 a non-zero value if the read fails or if the value will not fit
112 in DEST.
113
114 Note that DEST may not be large enough to hold all of the requested
115 data, but the function will read BYTES characters anyway.
116
117 BYTES may be a maximum of 10. */
118
119static int
120__read_long (long *dest, int file, size_t bytes)
121{
122 char c[10];
123
124 if (bytes > 10 || read(file, c, bytes) != bytes)
125 return 1;
126 else
127 return __fetch_long (dest, c, bytes);
128}
