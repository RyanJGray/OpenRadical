74#ifndef _NET_IF_MEDIA_H_
75#define _NET_IF_MEDIA_H_
76
77/*
78 * if_media Options word:
79 * Bits Use
80 * ---- -------
81 * 0-4 Media subtype MAX SUBTYPE == 31!
82 * 5-7 Media type
83 * 8-15 Type specific options
84 * 16-19 RFU
85 * 20-27 Shared (global) options
86 * 28-31 Status
87 */
88
89/*
90 * Ethernet
91 */
92#define IFM_ETHER 0x00000020
93#define IFM_10_T 3 /* 10BaseT - RJ45 */
94#define IFM_100_TX 6 /* 100BaseTX - RJ45 */
95#define IFM_1000_TX 16 /* 1000baseTX - 4 pair cat 5 */
96
97/*
98 * Shared media sub-types
99 */
100#define IFM_AUTO 0 /* Autoselect best media */
101#define IFM_MANUAL 1 /* Jumper/dipswitch selects media */
102#define IFM_NONE 2 /* Deselect all media */
103
104/*
105 * Shared options
106 */
107#define IFM_FDX 0x00100000 /* Force full duplex */
108#define IFM_HDX 0x00200000 /* Force half duplex */
109#define IFM_FLOW 0x00400000 /* enable hardware flow control */
110
111/*
112 * Status bits
113 */
114#define IFM_AVALID 0x10000000 /* Active bit valid */
115#define IFM_ACTIVE 0x20000000 /* Interface attached to working net */
116
117/*
118 * Masks
119 */
120#define IFM_NMASK 0x000000e0 /* Network type */
121#define IFM_TMASK 0x0000001f /* Media sub-type */
122#define IFM_SMASK 0xf0000000 /* Status */
123#define IFM_OMASK 0x0000ff00 /* Type specific options */
124#define IFM_GMASK 0x0ff00000 /* Global options */
125
126/*
127 * Macros to extract various bits of information from the media word.
128 */
129#define IFM_TYPE(x) ((x) & IFM_NMASK)
130#define IFM_SUBTYPE(x) ((x) & IFM_TMASK)
131#define IFM_STATUS(x) ((x) & IFM_SMASK)
132#define IFM_OPTIONS(x) ((x) & (IFM_OMASK|IFM_GMASK))
133
134#endif /* _NET_IF_MEDIA_H_ */
