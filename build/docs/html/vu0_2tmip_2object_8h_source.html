83// メンバ 内容

87// 640x480 モードでは (ax,ay)=(1.0,1.0) を指定しますが、
88// 640x240 モードでは (ax,ay)=(1.0,0.5) となります。
89// cx, cy 中心値。現在未使用
90// zmin,zmax Z バッファの値の最小／最大値。現在未使用
91// nearz,farz ビュー座標系での最近値と最遠値。現在未使用

93// scrz,ax,ay,cx,cy,zmin,zmax,nearz,farz より計算されます
94// vc ビュー座標系からクリップ座標系へ変換するための行列。
95// 現在未使用
96//
97// Screen property
98typedef struct {
99 float scrz; // distance to screen
100 float ax, ay; // aspect ratio
101 float cx, cy; // center point
102 float zmin, zmax; // Z-buffer range
103 float nearz, farz; // z edge
104 MATRIX vs; // view-screen matrix
105 MATRIX vc; // view-clip matrix
106} SCREEN;

115// ビュー座標系のプロパティを定義します。
116//
117// メンバ 内容
118//------------------------------------------------------------------------
119// yd ワールド座標系で見たカメラの脚の方向を示します。
120// カメラが地面に並行に置かれていれば (0,0,-1,0) となります
121// zd ワールド座標系で見たカメラの視線方向を示します
122// カメラが常に原点 (0,0,0) を向くようにするには p をカメ
123// ラ位置として、zd = (-p[0], -p[1], -p[2], 0) とします。
124// p ワールド座標系で見たカメラの位置を示します。
125// wv ワールド座標系からビュー座標系への変換行列。
126// yd, zd, p より計算されます。
127// ws ワールド座標系からスクリーン座標系への変換行列。
128// ローカル関数 updateView() で計算されます。
129// wc ワールド座標系からクリップ座標系への変換行列。
130// ローカル関数 updateView() で計算されます。
131// Your position property
132typedef struct {
133 VECTOR yd; // vertical vector (in the world)
134 VECTOR zd; // eye direction (in the world)
135 VECTOR p; // standing position
136 MATRIX wv; // world-view matrix
137 MATRIX ws; // world-screen matrix
138 MATRIX wc; // world-clip matrix
139} VIEW;

141// ---------------------------------------------------------------------------
142//
143// LIGHT
144//
145// ---------------------------------------------------------------------------
146// 光源計算のためのプロパティを定義します。
147// 光源は並行光源で 3 つ、点光源で 1 つを持つことができます。
148//
149// メンバ 内容
150//------------------------------------------------------------------------
151// model 光源モデルを指定します。これはオブジェクトハンドライン
152// タンス関数内で参照されることが期待されます。当然このフィー
153// ルドを参照しないインスタンス関数には無効です。
154// 1 ライティング計算を行ないません。
155// 2 並行光源
156// 3 疑似点光源。
157// オブジェクトの代表値と光源の位置から光源ベクトルを求
158// め、そのオブジェクトに対しては並行光源として扱います。
159// 3 点光源。減衰なしの点光源計算を全ての頂点に対して行な
160// います。
161//
162// lp0,lp1,lp2 ワールド座標系で示された光源の位置。点光源にのみ有効。
163// ld0,ld1,ld2 ワールド座標系で示された光源方向 並行光源にのみ有効。
164// c0, c1, c2 光源色
165// a 環境色。すべての光源が off であっても環境色が加算されます
166// nl ワールド座標系で記述されたノーマルライト行列
167// lc ライトカラー行列
168
169// lighting property
170typedef struct {
171 int model; // light model (0:off, 1:pararell, 2:pseudo spot, 3:spot
172 VECTOR lp0, lp1, lp2; // light source position (in the world)
173 VECTOR ld0, ld1, ld2; // light direction (in the world)
174 VECTOR c0, c1, c2; // light influence
175 VECTOR a; // ambient color
176 MATRIX nl; // world normal-light
177 MATRIX lc; // light-color matrix
178} LIGHT;

180// ---------------------------------------------------------------------------
181//
182// OBJECT
183//
184// ---------------------------------------------------------------------------
185// ３Ｄオブジェクトのプロパティを定義します。
186// OBJECT 構造体を介することで複数のオブジェクトドライバ（インスタン
187// ス関数）を統一的に操作することができます。
188// 各プロパティは、 構造体の driver メンバとして定義されるインスタンス
189// 関数内で参照されることが期待されます。インスタンス関数が値を参照し
190// ない場合は無効になります。
191//
192// メンバ 内容
193//------------------------------------------------------------------------
194// tag オブジェクトを構成するプリミティブに対する GIFtag
195// ローカル関数 ObjMakeTag() で指定します。
196// view 現在のビュー
197// light 現在のライト
198// mw モデル（ローカル）座標系からワールド座標系への変換行列
199// shadow 1 が指定されると z = 0 の面に対するドロップシャドウを
200// 行ないます。
201// clip 1 が指定されるとプリミティブ単位で画面クリップが行なわ
202// れます。
203// spec スペキュラ。1 以上を指定すると金属的な光沢が表現されます。
204// rate 内挿比率。二つの頂点（頂点と対頂点）を線形内挿して頂点
205// を生成する場合にその率を指定します。
206// c, c1 頂点と対頂点のデフォルト色を指定します。頂点色はこれを
207// rate で線形内挿した値がとられます
208// cvnt color-vertex-normal-texture データ。
209// 頂点データは GIFtag に続くストリップメッシュが期待され
210// ます。メッシュの個数は GIFtag の NLOOP フィールドから
211// 取得されます。つまりオブジェクトデータは、
212//
213// GIFtag
214// (R, G, B, A) が N 個続く
215// (Vx,Vy,Vz,1) が N 個続く
216// (Nx,Ny,Nz,1) が N 個続く
217// (S, T, Q, 1) が N 個続く
218//
219// の形で記述されているものとみなされます。
220// GIFtag のフィールドは tag メンバで NLOOP を除きオーバー
221// ライトされます
222// cvnt1 対頂点。内挿を使用しない場合は無効です。
223// box ラウンディングボックス。box[0]-box[7] までの 8 頂点で
224// オブジェクトを包絡する８面体を定義します。これはオブジェ
225// dirver[13] Path1/Path3 を経由してディスプレイリストを作るためのイ
226// ンスタンス関数。両方を指定することができます。
227//
228// object handler
229typedef struct {
230 qword tag; // GIFtag (path3) REFtag (path1)
231 VIEW *view; // view
232 LIGHT *light; // light
233 MATRIX *mw; // model-world matrix
234 u_char shadow; // indicate shadow (to Z=0 plane)
235 u_char clip; // clip enable
236 u_char spec; // specular
237 float rate; // interpolate rate
238 VECTOR *c; // fixed color
239 VECTOR *c1; // counter fixed color
240 VECTOR *cvnt; // color-vertex-normal-texture
241 VECTOR *cvnt1; // countter color-vertex-normal-texture
242 VECTOR *box; // rounding box
243 qword *(*driver1)(); // path1 driver pointer
244 qword *(*driver3)(); // path3 driver pointer
245} OBJECT;
