120/*
121 * TCP statistics.
122 * Many of these should be kept per connection,
123 * but that's inconvenient at the moment.
124 */
125struct sceEENetTcpstat {
126 u_quad_t tcps_connattempt; /* connections initiated */
127 u_quad_t tcps_accepts; /* connections accepted */
128 u_quad_t tcps_connects; /* connections established */
129 u_quad_t tcps_drops; /* connections dropped */
130 u_quad_t tcps_conndrops; /* embryonic connections dropped */
131 u_quad_t tcps_closed; /* conn. closed (includes drops) */
132 u_quad_t tcps_segstimed; /* segs where we tried to get rtt */
133 u_quad_t tcps_rttupdated; /* times we succeeded */
134 u_quad_t tcps_delack; /* delayed acks sent */
135 u_quad_t tcps_timeoutdrop; /* conn. dropped in rxmt timeout */
136 u_quad_t tcps_rexmttimeo; /* retransmit timeouts */
137 u_quad_t tcps_persisttimeo; /* persist timeouts */
138 u_quad_t tcps_keeptimeo; /* keepalive timeouts */
139 u_quad_t tcps_keepprobe; /* keepalive probes sent */
140 u_quad_t tcps_keepdrops; /* connections dropped in keepalive */
141 u_quad_t tcps_persistdrops; /* connections dropped in persist */
142 u_quad_t tcps_connsdrained; /* connections drained due to memory
143 shortage */
144 u_quad_t tcps_pmtublackhole; /* PMTUD blackhole detected */
145
146 u_quad_t tcps_sndtotal; /* total packets sent */
147 u_quad_t tcps_sndpack; /* data packets sent */
148 u_quad_t tcps_sndbyte; /* data bytes sent */
149 u_quad_t tcps_sndrexmitpack; /* data packets retransmitted */
150 u_quad_t tcps_sndrexmitbyte; /* data bytes retransmitted */
151 u_quad_t tcps_sndacks; /* ack-only packets sent */
152 u_quad_t tcps_sndprobe; /* window probes sent */
153 u_quad_t tcps_sndurg; /* packets sent with URG only */
154 u_quad_t tcps_sndwinup; /* window update-only packets sent */
155 u_quad_t tcps_sndctrl; /* control (SYN|FIN|RST) packets sent */
156
157 u_quad_t tcps_rcvtotal; /* total packets received */
158 u_quad_t tcps_rcvpack; /* packets received in sequence */
159 u_quad_t tcps_rcvbyte; /* bytes received in sequence */
160 u_quad_t tcps_rcvbadsum; /* packets received with ccksum errs */
161 u_quad_t tcps_rcvbadoff; /* packets received with bad offset */
162 u_quad_t tcps_rcvmemdrop; /* packets dropped for lack of memory */
163 u_quad_t tcps_rcvshort; /* packets received too short */
164 u_quad_t tcps_rcvduppack; /* duplicate-only packets received */
165 u_quad_t tcps_rcvdupbyte; /* duplicate-only bytes received */
166 u_quad_t tcps_rcvpartduppack; /* packets with some duplicate data */
167 u_quad_t tcps_rcvpartdupbyte; /* dup. bytes in part-dup. packets */
168 u_quad_t tcps_rcvoopack; /* out-of-order packets received */
169 u_quad_t tcps_rcvoobyte; /* out-of-order bytes received */
170 u_quad_t tcps_rcvpackafterwin; /* packets with data after window */
171 u_quad_t tcps_rcvbyteafterwin; /* bytes rcvd after window */
172 u_quad_t tcps_rcvafterclose; /* packets rcvd after "close" */
173 u_quad_t tcps_rcvwinprobe; /* rcvd window probe packets */
174 u_quad_t tcps_rcvdupack; /* rcvd duplicate acks */
175 u_quad_t tcps_rcvacktoomuch; /* rcvd acks for unsent data */
176 u_quad_t tcps_rcvackpack; /* rcvd ack packets */
177 u_quad_t tcps_rcvackbyte; /* bytes acked by rcvd acks */
178 u_quad_t tcps_rcvwinupd; /* rcvd window update packets */
179 u_quad_t tcps_pawsdrop; /* segments dropped due to PAWS */
180 u_quad_t tcps_predack; /* times hdr predict ok for acks */
181 u_quad_t tcps_preddat; /* times hdr predict ok for data pkts */
182
183 u_quad_t tcps_pcbhashmiss; /* input packets missing pcb hash */
184 u_quad_t tcps_noport; /* no socket on port */
185 u_quad_t tcps_badsyn; /* received ack for which we have
186 no SYN in compressed state */
187};
