127// make view_screen & view_clip & clip_screen matrix
128void SetViewScreenClipMatrix(sceVu0FMATRIX m, sceVu0FMATRIX mc,
129 sceVu0FMATRIX mcs,
130 float scrz, float ax, float ay,
131 float cx, float cy,
132 float zmin, float zmax, float nearz, float farz)
133{
134 float az, cz;
135 float gsx, gsy;
136 sceVu0FMATRIX mt;
137
138 gsx = nearz*clip_volume[0]/scrz;
139 gsy = nearz*clip_volume[1]/scrz;
140
141 cz = (-zmax * nearz + zmin * farz) / (-nearz + farz);
142 az = farz * nearz * (-zmin + zmax) / (-nearz + farz);
143
144 // | scrz 0 0 0 |
145 // m = | 0 scrz 0 0 |
146 // | 0 0 0 1 |
147 // | 0 0 1 0 |
148 sceVu0UnitMatrix(m);
149 m[0][0] = scrz; m[1][1] = scrz;
150 m[2][2] = 0; m[3][3] = 0;
151 m[3][2] = 1.0; m[2][3] = 1.0;
152
153 // | ax 0 0 cx |
154 // m = | 0 ay 0 cy |
155 // | 0 0 az cz |
156 // | 0 0 0 1 |
157 sceVu0UnitMatrix(mt);
158 mt[0][0] = ax; mt[1][1] = ay; mt[2][2] = az;
159 mt[3][0] = cx; mt[3][1] = cy; mt[3][2] = cz;
160 sceVu0MulMatrix(m, mt, m);
161
162
163 // |2n/2gsx 0 0 0 |
164 // m = | 0 2n/2gsy 0 0 |
165 // | 0 0 (f+n)/(f-n) -2f*n/(f-n) |
166 // | 0 0 1 0 |
167 sceVu0UnitMatrix(mc);
168 mc[0][0] = 2 * nearz / (gsx-(-gsx));
169 mc[1][1] = 2 * nearz / (gsy-(-gsy));
170 mc[2][2] = (farz + nearz) / (farz-nearz);
171 mc[3][2] = -2*(farz * nearz) / (farz-nearz);
172 mc[2][3] = 1; mc[3][3] = 0;
173
174 // |scrz*ax*gsx/n 0 0 cx |
175 // m = | 0 scrz*ay*gsy/n 0 cy |
176 // | 0 0 (-zmax+zmin)/2 (zmin+zmax)/2 |
177 // | 0 0 0 1 |
178 sceVu0UnitMatrix(mcs);
179 mcs[0][0] = scrz*ax*gsx/nearz;
180 mcs[1][1] = scrz*ay*gsy/nearz;
181 mcs[2][2] = (-zmax+zmin)/2;
182 mcs[3][2] = (zmax+zmin)/2;
183 mcs[3][0] = cx; mcs[3][1] = cy;
184 //mcs[3][2] = cz;
185 mcs[3][3] = 1;
186
187 return;
188}
