131// make view_screen & view_clip matrix
132void SetViewScreenClipMatrix(sceVu0FMATRIX m, sceVu0FMATRIX mc,
133 sceVu0FMATRIX mcs,
134 float scrz, float ax, float ay,
135 float cx, float cy,
136 float zmin, float zmax, float nearz, float farz)
137{
138 float az, cz;
139 float gsx, gsy;
140 sceVu0FMATRIX mt;
141
142 gsx = nearz*clip_volume[0]/scrz;
143 gsy = nearz*clip_volume[1]/scrz;
144
145 cz = (-zmax * nearz + zmin * farz) / (-nearz + farz);
146 az = farz * nearz * (-zmin + zmax) / (-nearz + farz);
147
148 // | scrz 0 0 0 |
149 // m = | 0 scrz 0 0 |
150 // | 0 0 0 1 |
151 // | 0 0 1 0 |
152 sceVu0UnitMatrix(m);
153 m[0][0] = scrz; m[1][1] = scrz;
154 m[2][2] = 0; m[3][3] = 0;
155 m[3][2] = 1.0; m[2][3] = 1.0;
156
157 // | ax 0 0 cx |
158 // m = | 0 ay 0 cy |
159 // | 0 0 az cz |
160 // | 0 0 0 1 |
161 sceVu0UnitMatrix(mt);
162 mt[0][0] = ax; mt[1][1] = ay; mt[2][2] = az;
163 mt[3][0] = cx; mt[3][1] = cy; mt[3][2] = cz;
164
165 sceVu0MulMatrix(m, mt, m);
166
167 // |2n/2gsx 0 0 0 |
168 // m = | 0 2n/2gsy 0 0 |
169 // | 0 0 (f+n)/(f-n) -2f*n/(f-n) |
170 // | 0 0 1 0 |
171 sceVu0UnitMatrix(mc);
172 mc[0][0] = 2 * nearz / (gsx-(-gsx));
173 mc[1][1] = 2 * nearz / (gsy-(-gsy));
174 mc[2][2] = (farz + nearz) / (farz-nearz);
175 mc[3][2] = -2*(farz * nearz) / (farz-nearz);
176 mc[2][3] = 1; mc[3][3] = 0;
177
178 // |scrz*ax*gsx/n 0 0 cx |
179 // m = | 0 scrz*ay*gsy/n 0 cy |
180 // | 0 0 (-zmax+zmin)/2 (zmin+zmax)/2 |
181 // | 0 0 0 1 |
182 sceVu0UnitMatrix(mcs);
183 mcs[0][0] = scrz*ax*gsx/nearz;
184 mcs[1][1] = scrz*ay*gsy/nearz;
185 mcs[2][2] = (-zmax+zmin)/2;
186 mcs[3][2] = (zmax+zmin)/2;
187 mcs[3][0] = cx; mcs[3][1] = cy;
188 //mcs[3][2] = cz;
189 mcs[3][3] = 1;
190
191 return;
192}
